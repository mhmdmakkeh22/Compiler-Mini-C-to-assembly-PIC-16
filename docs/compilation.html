 60class Token:
 61    def __init__(self, type_, value, line, col):
 62        self.type = type_
 63        self.value = value
 64        self.line = line
 65        self.col = col
 66    def __repr__(self):
 67        return f"Token({self.type}, {self.value}, line={self.line}, col={self.col})"
 68
 69class MiniCLexer:
 70    def __init__(self, code):
 71        self.code = code
 72        parts = []
 73        for idx, (pattern, name) in enumerate(token_specification):
 74            group = name or f"SKIP_{idx}"
 75            parts.append(f"(?P<{group}>{pattern})")
 76        self.big_regex = re.compile('|'.join(parts))
 77
 78    def tokenize(self):
 79        tokens = []
 80        line, col = 1, 1
 81        for match in self.big_regex.finditer(self.code):
 82            kind = match.lastgroup
 83            text = match.group()
 84            # Skip whitespace/comments
 85            if kind and kind.startswith('SKIP'):
 86                for ch in text:
 87                    if ch == '\n': line += 1; col = 1
 88                    else: col += 1
 89                continue
 90            tokens.append(Token(kind, text, line, col))
 91            for ch in text:
 92                if ch == '\n': line += 1; col = 1
 93                else: col += 1
 94        return tokens

173# === Modified Parser implementation ===
174class ParsingException(Exception):
175    pass
176
177class Parser:
178    def __init__(self, tokens):
179        self.tokens = tokens
180        self.index = 0
181
182    def peek(self):
183        return self.tokens[self.index] if self.index < len(self.tokens) else None
184
185    def peek_type(self):
186        tk = self.peek()
187        return tk.type if tk else None
188
189    def accept(self, expected=None):
190        tk = self.peek()
191        if not tk:
192            raise ParsingException(f"Unexpected end, wanted {expected}")
193        if expected and tk.type != expected:
194            raise ParsingException(f"Expected {expected}, got {tk.type}")
195        self.index += 1
196        return tk
197
198    def expect(self, t):
199        return self.accept(t)
200
201    def finished(self):
202        return self.index >= len(self.tokens)
203
204    def error(self, msg):
205        tk = self.peek()
206        if tk:
207            raise ParsingException(
208                f"Syntax error at line {tk.line}, col {tk.col}: {msg}, "
209                f"found {tk.type} -> '{tk.value}'"
210            )
211        else:
212            raise ParsingException(f"Syntax error at end of input: {msg}")
213
214    # Top-level entry point
215    def parse(self):
216        program = self.parse_program()
217        if not self.finished():
218            self.error("Extra tokens after parsing complete.")
219        return program
220
221    # Modified program parsing to handle both main() and standalone code
222    # program â†’ (int main() { declaration* statement* }) | (declaration* statement*)
223    def parse_program(self):
224        # Check if the program starts with the main function
225        if (self.peek_type() == 'KW_INT' and 
226            len(self.tokens) > self.index + 1 and 
227            self.tokens[self.index + 1].type == 'KW_MAIN'):
228            # Parse as traditional main function
229            self.expect('KW_INT')
230            self.expect('KW_MAIN')
231            self.expect('LPAREN')
232            self.expect('RPAREN')
233            self.expect('LBRACE')
234
235            decls = []
236            while self.peek_type() in ('KW_INT','KW_FLOAT','KW_BOOL','KW_CHAR'):
237                decls.append(self.parse_declaration())
238
239            stmts = []
240            while self.peek_type() not in (None,'RBRACE'):
241                stmts.append(self.parse_statement())
242
243            self.expect('RBRACE')
244        else:
245            # Parse as standalone code without main function
246            decls = []
247            while self.peek_type() in ('KW_INT','KW_FLOAT','KW_BOOL','KW_CHAR'):
248                decls.append(self.parse_declaration())
249
250            stmts = []
251            while self.peek_type() not in (None,):
252                stmts.append(self.parse_statement())
253
254        return Program(decls, stmts)
255
256    def parse_declaration(self):
257        var_type = self.parse_type()
258        name = self.accept('IDENT').value
259
260        array_size = None
261        if self.peek_type() == 'LBRACKET':
262            self.accept('LBRACKET')
263            array_size = int(self.accept('INT_LITERAL').value)
264            self.expect('RBRACKET')
265
266        self.expect('SEMICOLON')
267        return Declaration(var_type, name, array_size)
268
269    def parse_type(self):
270        t = self.peek_type()
271        if t not in ('KW_INT','KW_FLOAT','KW_BOOL','KW_CHAR'):
272            self.error(f"Expected type, got {t}")
273        # consume the keyword and map to a string
274        mapping = {
275            'KW_INT':'int', 'KW_FLOAT':'float',
276            'KW_BOOL':'bool','KW_CHAR':'char'
277        }
278        return mapping[self.accept(t).type]
279
280    def parse_statement(self):
281        pt = self.peek_type()
282        if   pt == 'IDENT':
283            return self.parse_assignment()
284        elif pt == 'KW_IF':
285            return self.parse_if()
286        elif pt == 'KW_WHILE':
287            return self.parse_while()
288        elif pt == 'LBRACE':
289            return self.parse_block()
290        else:
291            self.error(f"Expected statement, got {pt}")
292
293    def parse_assignment(self):
294        name = self.accept('IDENT').value
295
296        idx = None
297        if self.peek_type() == 'LBRACKET':
298            self.accept('LBRACKET')
299            idx = self.parse_expression()
300            self.expect('RBRACKET')
301
302        self.expect('ASSIGN')
303        expr = self.parse_expression()
304        self.expect('SEMICOLON')
305        return AssignmentStatement(name, idx, expr)
306
307    def parse_if(self):
308        self.accept('KW_IF'); self.accept('LPAREN')
309        cond = self.parse_expression()
310        self.expect('RPAREN')
311        then_blk = self.parse_block()
312
313        else_blk = None
314        if self.peek_type() == 'KW_ELSE':
315            self.accept('KW_ELSE')
316            else_blk = self.parse_block()
317
318        return IfStatement(cond, then_blk, else_blk)
319
320    def parse_while(self):
321        self.accept('KW_WHILE'); self.accept('LPAREN')
322        cond = self.parse_expression()
323        self.expect('RPAREN')
324        body = self.parse_block()
325        return WhileStatement(cond, body)
326
327    def parse_block(self):
328        self.accept('LBRACE')
329        decls = []
330        while self.peek_type() in ('KW_INT','KW_FLOAT','KW_BOOL','KW_CHAR'):
331            decls.append(self.parse_declaration())
332        stmts = []
333        while self.peek_type() not in (None,'RBRACE'):
334            stmts.append(self.parse_statement())
335        self.expect('RBRACE')
336        return Block(decls, stmts)
337
338    # Expression grammar with correct precedence
339    def parse_expression(self):
340        return self.parse_or()
341    def parse_or(self):
342        left = self.parse_and()
343        while self.peek_type() == 'OR':
344            op    = self.accept().value
345            right = self.parse_and()
346            left  = BinaryOp(op, left, right)
347        return left
348    def parse_and(self):
349        left = self.parse_equality()
350        while self.peek_type() == 'AND':
351            op    = self.accept().value
352            right = self.parse_equality()
353            left  = BinaryOp(op, left, right)
354        return left
355    def parse_equality(self):
356        left = self.parse_relational()
357        while self.peek_type() in ('EQ','NEQ'):
358            op    = self.accept().value
359            right = self.parse_relational()
360            left  = BinaryOp(op, left, right)
361        return left
362    def parse_relational(self):
363        left = self.parse_additive()
364        while self.peek_type() in ('LT','GT','LTE','GTE'):
365            op    = self.accept().value
366            right = self.parse_additive()
367            left  = BinaryOp(op, left, right)
368        return left
369    def parse_additive(self):
370        left = self.parse_multiplicative()
371        while self.peek_type() in ('PLUS','MINUS'):
372            op    = self.accept().value
373            right = self.parse_multiplicative()
374            left  = BinaryOp(op, left, right)
375        return left
376    def parse_multiplicative(self):
377        left = self.parse_unary()
378        while self.peek_type() in ('MULT','DIV'):
379            op    = self.accept().value
380            right = self.parse_unary()
381            left  = BinaryOp(op, left, right)
382        return left
383    def parse_unary(self):
384        if self.peek_type() in ('MINUS','NOT'):
385            op   = self.accept().value
386            expr = self.parse_unary()
387            return UnaryOp(op, expr)
388        return self.parse_primary()
389    def parse_primary(self):
390        pt = self.peek_type()
391        if pt == 'IDENT':
392            tok = self.accept('IDENT')
393            idx = None
394            if self.peek_type() == 'LBRACKET':
395                self.accept('LBRACKET')
396                idx = self.parse_expression()
397                self.expect('RBRACKET')
398            return Identifier(tok.value, idx)
399        elif pt == 'INT_LITERAL':
400            tok = self.accept('INT_LITERAL')
401            return Literal(int(tok.value))
402        elif pt == 'LPAREN':
403            self.accept('LPAREN')
404            expr = self.parse_expression()
405            self.expect('RPAREN')
406            return Parenthesized(expr)
407        else:
408            self.error(f"Unexpected primary {pt}")

412class Visitor: pass
413class CodeGenVisitor(Visitor):
414    def __init__(self):
415        self.code = []
416        self.var_map = {}
417        self.next_addr = 0x20
418        self.label_counter = 0
419    def emit(self, line): self.code.append(line)
420    def make_label(self,prefix="lbl"): lbl=f"{prefix}{self.label_counter}"; self.label_counter+=1; return lbl
421    def get_code(self): return "\n".join(self.code)
422    def alloc_var(self,name):
423        if name not in self.var_map:
424            addr = self.next_addr; self.var_map[name] = f"0x{addr:02X}"; self.next_addr+=1
425        return self.var_map[name]
426    def visitProgram(self,node):
427        for d in node.declarations: d.accept(self)
428        for s in node.statements: s.accept(self)
429    def visitBlock(self,node):
430        for d in node.declarations: d.accept(self)
431        for s in node.statements: s.accept(self)
432    def visitDeclaration(self,node): self.alloc_var(node.name)
433    def visitAssignment(self,node):
434        node.rhs.accept(self)
435        addr=self.alloc_var(node.name)
436        if node.index_expr: self.emit("; array indexing not implemented")
437        self.emit(f"MOVWF {addr}")
438    def visitIf(self,node):
439        else_lbl=self.make_label("else")
440        end_lbl=self.make_label("ifend")
441        node.condition.accept(self)
442        self.emit("CPFSEQ W")
443        self.emit(f"GOTO {else_lbl}")
444        node.then_block.accept(self)
445        self.emit(f"GOTO {end_lbl}")
446        self.emit(f"{else_lbl}:")
447        if node.else_block: node.else_block.accept(self)
448        self.emit(f"{end_lbl}:")
449    def visitWhile(self,node):
450        top_lbl=self.make_label("while")
451        end_lbl=self.make_label("wend")
452        self.emit(f"{top_lbl}:")
453        node.condition.accept(self)
454        self.emit("CPFSEQ W")
455        self.emit(f"GOTO {end_lbl}")
456        node.body.accept(self)
457        self.emit(f"GOTO {top_lbl}")
458        self.emit(f"{end_lbl}:")
459    def visitBinaryOp(self,node):
460        node.left.accept(self)
461        self.emit("MOVWF 0x7F")
462        node.right.accept(self)
463        if node.op == "+": self.emit("ADDWF 0x7F, W")
464        elif node.op == "-": self.emit("SUBWF 0x7F, W")
465        elif node.op == "*": self.emit("; MULT not implemented")
466        elif node.op == "/": self.emit("; DIV not implemented")
467        else: self.emit(f"; op {node.op} not implemented")
468    def visitUnaryOp(self,node):
469        node.expr.accept(self)
470        if node.op == "-": self.emit("; unary minus not implemented")
471        elif node.op == "!": self.emit("; logical not not implemented")
472    def visitLiteral(self,node):
473        val=node.value & 0xFF; self.emit(f"MOVLW 0x{val:02X}")
474    def visitIdentifier(self,node): self.emit(f"MOVF {self.alloc_var(node.name)}, W")
475    def visitParenthesized(self,node): node.expr.accept(self)
